SELECT TOP 10
  title, 
  CAST(SUBSTRING(duration, 1, LEN(duration) - 4) AS INT) AS duration_in_minutes
FROM 
    titles
WHERE 
    duration LIKE '% min' 
ORDER BY 
    duration_in_minutes DESC;
